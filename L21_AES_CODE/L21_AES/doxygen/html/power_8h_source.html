804 static inline enum status_code system_switch_performance_level(
805  const enum system_performance_level performance_level)
806 {
807 
808  if (performance_level == (enum system_performance_level)PM->PLCFG.reg) {
809  return STATUS_OK;
810  }
811 
812 #if SAML22 || SAML21XXXB
813  if (PM->PLCFG.reg & PM_PLCFG_PLDIS) {
814  return STATUS_ERR_INVALID_ARG;
815  }
816 #endif
817 
818  /* Clear performance level status */
819  PM->INTFLAG.reg = PM_INTFLAG_PLRDY;
820 
821  /* Switch performance level */
822  PM->PLCFG.reg = performance_level;
823 
824  /* Waiting performance level ready */
825  while (!PM->INTFLAG.reg) {
826  ;
827  }
828  return STATUS_OK;
829 }
